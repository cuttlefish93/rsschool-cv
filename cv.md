### Code Examples:

**Codewars** - Give me a Diamond, _6kyu_

```
function diamond(n){
  if(n <= 0 || n % 2 === 0) {
    return null;
  }

  let diamond = '*';
  let newlineChar = '\n';
  let str = '';

  for(let i = 1; i <= n; i += 2){
    diamond = diamond.repeat(i);
    str = `${str}${' '.repeat((n - i) / 2)}${diamond}${newlineChar}`;
    diamond = '*';
  }

  for(let i = n - 2; i >= 1; i -= 2){
    diamond = diamond.repeat(i);
    str = `${str}${' '.repeat((n - i) / 2)}${diamond}${newlineChar}`;
    diamond = '*';
  }
  return str;
}
```
